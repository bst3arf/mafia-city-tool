function toggleLang(l) {
   document.documentElement.dir = (l === 'ar' ? 'rtl' : 'ltr');
   if(l === 'en') document.getElementById('app').classList.add('en');
   else document.getElementById('app').classList.remove('en');
   for(let key in ui[l]) document.getElementById(key).innerText = ui[l][key];
   document.documentElement.lang = l;
}

function updateHeroInfo() {
   const h = document.getElementById('hero1').value;
   const info = document.getElementById('heroInfo');
   const lang = document.documentElement.lang || 'ar';
   if(h === 'none') info.style.display = 'none';
   else {
       info.style.display = 'block';
       info.innerHTML = `<b>💡 ${lang=='ar'?'ملاحظة البطل':'Hero Note'}:</b><br>${heroLib[h][lang]}`;
   }
}

function calculate() {
   const size = parseInt(document.getElementById('marchSize').value);
   const event = document.getElementById('event').value;
   const lang = document.documentElement.lang || 'ar';
   
   let pInf = 0.30, pBike = 0.65, pCar = 0.05;
   if(event === 'defense') { pInf = 0.45; pBike = 0.45; pCar = 0.10; }

   const armyHTML = `
       <div class="army-row">${lang=='ar'?'مقاتلين':'Bulkers'} <span>${Math.floor(size*pInf).toLocaleString()}</span></div>
       <div class="army-row">${lang=='ar'?'دراجات':'Bikers'} <span>${Math.floor(size*pBike).toLocaleString()}</span></div>
       <div class="army-row">${lang=='ar'?'رماة/سيارات':'Shooters/Cars'} <span>${Math.floor(size*pCar).toLocaleString()}</span></div>
   `;
   
   document.getElementById('armyList').innerHTML = armyHTML;
   document.getElementById('finalTip').innerHTML = lang === 'ar'
       ? "⚠️ <b>نصيحة:</b> إذا كان الخصم يستخدم 'رماة' بكثرة، تأكد من زيادة نسبة الدراجات لـ 75%."
       : "⚠️ <b>Tip:</b> If opponent uses many 'Shooters', increase Biker ratio to 75%.";
   
   document.getElementById('results').style.display = 'block';
   document.getElementById('shareBtn').style.display = 'flex';
}

function share() {
   const size = document.getElementById('marchSize').value;
   const text = `🛡️ تشكيلة مافيا سيتي الاحترافية:\nسعة المسيرة: ${size}\nافتح الرابط لرؤية التشكيلة كاملة!`;
   window.open(`https://wa.me/?text=${encodeURIComponent(text)}`);
}
